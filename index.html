const express = require('express'); const cors = require('cors'); const mercadopago = require('mercadopago'); const app = express(); app.use(cors()); app.use(express.json()); // Configuración de Mercado Pago mercadopago.configure({ access_token: process.env.MP_ACCESS_TOKEN }); app.get('/', (req, res) => res.send('Rixa backend OK 🚀')); app.post('/crear-preferencia', async (req, res) => { try { const preferencia = { items: req.body.items, back_urls: { success: process.env.FRONTEND_URL, failure: process.env.FRONTEND_URL, pending: process.env.FRONTEND_URL }, auto_return: "approved", }; const respuesta = await mercadopago.preferences.create(preferencia); res.json({ id: respuesta.body.id }); } catch (error) { console.error(error); res.status(500).json({ error: "Error al crear preferencia" }); } }); app.listen(process.env.PORT || 8080, () => console.log('Servidor corriendo'));
